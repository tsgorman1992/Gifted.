import React, { useState, useRef } from "react";
import { useLocation } from "wouter";
import { motion } from "framer-motion";
import { Loader2, X, Video, Plus } from "lucide-react";
import { Button } from "@/components/ui/button";
import { useAuth } from "@/lib/auth-context";
import { useUpload } from "@workspace/object-storage-web";
import { EXPERIENCE_LIST, DEFAULT_EXPERIENCE, getSuggestedExperience, type ExperienceId, gradientStyle } from "@/lib/experiences";

const BASE = (import.meta.env.BASE_URL ?? "/").replace(/\/+$/, "");
const OCCASIONS = ["Birthday", "Anniversary", "Love", "Graduation", "New Baby", "Holiday", "Just Because", "Wedding", "Thank You", "Mother's Day", "Father's Day", "Other"];
const MAX_PHOTOS = 6;
const MAX_PHOTO_SIZE = 20 * 1024 * 1024;
const ACCEPTED_PHOTO_TYPES = ["image/jpeg", "image/png", "image/webp", "image/heic"];

interface PhotoItem {
  id: string;
  objectPath: string;
  previewUrl: string;
}

export default function ChipInCreatePage() {
  const { isAuthenticated, isLoading: authLoading, user } = useAuth();
  const [, setLocation] = useLocation();

  const [organizerName, setOrganizerName] = useState("");
  const [organizerContributes, setOrganizerContributes] = useState(false);
  const [recipientName, setRecipientName] = useState("");
  const [recipientPhone, setRecipientPhone] = useState("");
  const [occasion, setOccasion] = useState("Birthday");
  const [giftTitle, setGiftTitle] = useState("");
  const [experience, setExperience] = useState<ExperienceId>(DEFAULT_EXPERIENCE);
  const [personalNote, setPersonalNote] = useState("");
  const [playlistUrl, setPlaylistUrl] = useState("");
  const [pitchMessage, setPitchMessage] = useState("");
  const [fixedAmount, setFixedAmount] = useState("");
  const [maxContributors, setMaxContributors] = useState("20");

  const [videoObjectPath, setVideoObjectPath] = useState<string | null>(null);
  const [videoPreviewUrl, setVideoPreviewUrl] = useState<string | null>(null);
  const videoBlobUrlRef = useRef<string | null>(null);
  const videoInputRef = useRef<HTMLInputElement>(null);
  const [videoError, setVideoError] = useState<string | null>(null);

  const [photos, setPhotos] = useState<PhotoItem[]>([]);
  const [photosUploading, setPhotosUploading] = useState<{ id: string; progress: number }[]>([]);
  const photoInputRef = useRef<HTMLInputElement>(null);
  const [photoError, setPhotoError] = useState<string | null>(null);

  const [submitting, setSubmitting] = useState(false);
  const [formError, setFormError] = useState<string | null>(null);

  const { uploadFile } = useUpload({
    basePath: `${BASE}/api/storage`,
    onSuccess: (response) => setVideoObjectPath(response.objectPath),
  });

  React.useEffect(() => {
    if (!authLoading && !isAuthenticated) {
      setLocation(`/sign-in?returnTo=${encodeURIComponent(window.location.pathname)}`);
    }
  }, [authLoading, isAuthenticated, setLocation]);

  React.useEffect(() => {
    if (user?.firstName && !organizerName) {
      setOrganizerName([user.firstName, user.lastName].filter(Boolean).join(" "));
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [user]);

  React.useEffect(() => {
    setExperience(getSuggestedExperience(occasion));
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [occasion]);

  async function handleVideoSelect(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0];
    if (!file) return;
    setVideoError(null);
    if (file.size > 200 * 1024 * 1024) {
      setVideoError("This video exceeds the 200 MB limit.");
      if (videoInputRef.current) videoInputRef.current.value = "";
      return;
    }
    if (!file.type.startsWith("video/")) {
      setVideoError("Please select a video file (MP4, MOV, etc.).");
      if (videoInputRef.current) videoInputRef.current.value = "";
      return;
    }
    if (videoBlobUrlRef.current) URL.revokeObjectURL(videoBlobUrlRef.current);
    const blobUrl = URL.createObjectURL(file);
    videoBlobUrlRef.current = blobUrl;
    setVideoPreviewUrl(blobUrl);
    const result = await uploadFile(file);
    if (videoInputRef.current) videoInputRef.current.value = "";
    if (!result) {
      setVideoObjectPath(null);
      setVideoPreviewUrl(null);
      setVideoError("Upload failed — please try again.");
    }
  }

  function handleRemoveVideo() {
    if (videoBlobUrlRef.current) { URL.revokeObjectURL(videoBlobUrlRef.current); videoBlobUrlRef.current = null; }
    setVideoObjectPath(null);
    setVideoPreviewUrl(null);
  }

  async function handlePhotoSelect(e: React.ChangeEvent<HTMLInputElement>) {
    const files = Array.from(e.target.files ?? []);
    if (!files.length) return;
    if (photoInputRef.current) photoInputRef.current.value = "";
    setPhotoError(null);
    const slotsAvailable = MAX_PHOTOS - photos.length - photosUploading.length;
    const toUpload = files.slice(0, slotsAvailable);
    if (toUpload.length < files.length) setPhotoError(`You can add up to ${MAX_PHOTOS} photos total.`);

    for (const file of toUpload) {
      if (file.size > MAX_PHOTO_SIZE) { setPhotoError(`"${file.name}" is over 20 MB.`); continue; }
      if (!ACCEPTED_PHOTO_TYPES.includes(file.type)) { setPhotoError(`"${file.name}" must be JPEG, PNG, WebP, or HEIC.`); continue; }
      const uploadId = crypto.randomUUID();
      setPhotosUploading(prev => [...prev, { id: uploadId, progress: 0 }]);
      try {
        const res = await fetch(`${BASE}/api/storage/uploads/request-url`, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ name: file.name, size: file.size, contentType: file.type }),
        });
        if (!res.ok) throw new Error("Failed to get upload URL");
        const { uploadURL, objectPath } = await res.json();
        await new Promise<void>((resolve, reject) => {
          const xhr = new XMLHttpRequest();
          xhr.open("PUT", uploadURL, true);
          xhr.setRequestHeader("Content-Type", file.type);
          xhr.upload.onprogress = (ev) => {
            if (ev.lengthComputable) {
              setPhotosUploading(prev => prev.map(p => p.id === uploadId ? { ...p, progress: Math.round((ev.loaded / ev.total) * 100) } : p));
            }
          };
          xhr.onload = () => (xhr.status >= 200 && xhr.status < 300 ? resolve() : reject(new Error("Upload failed")));
          xhr.onerror = () => reject(new Error("Upload failed"));
          xhr.send(file);
        });
        const servingUrl = `${BASE}/api/storage${objectPath}`;
        setPhotos(prev => [...prev, { id: uploadId, objectPath, previewUrl: servingUrl }]);
      } catch {
        setPhotoError(`Failed to upload "${file.name}".`);
      } finally {
        setPhotosUploading(prev => prev.filter(p => p.id !== uploadId));
      }
    }
  }

  function handleRemovePhoto(id: string) {
    setPhotos(prev => prev.filter(p => p.id !== id));
  }

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    setFormError(null);

    if (!organizerName.trim()) { setFormError("Your name is required."); return; }
    if (!recipientName.trim()) { setFormError("Recipient name is required."); return; }
    if (!giftTitle.trim()) { setFormError("Give this moment a title."); return; }
    if (!pitchMessage.trim()) { setFormError("Tell contributors what this is for."); return; }
    const amountNum = parseFloat(fixedAmount);
    if (!fixedAmount || isNaN(amountNum) || amountNum < 10 || amountNum > 1000) {
      setFormError("Each contribution must be between $10 and $1,000.");
      return;
    }

    setSubmitting(true);
    try {
      const res = await fetch(`${BASE}/api/gifted/group-gifts`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        credentials: "include",
        body: JSON.stringify({
          organizerName: organizerName.trim(),
          organizerContributes,
          recipientName: recipientName.trim(),
          recipientPhone: recipientPhone.trim() || undefined,
          occasion,
          giftTitle: giftTitle.trim(),
          experience,
          personalNote: personalNote.trim() || undefined,
          videoPath: videoObjectPath || undefined,
          photoPaths: photos.length > 0 ? photos.map(p => p.objectPath) : undefined,
          playlistUrl: playlistUrl.trim() || undefined,
          pitchMessage: pitchMessage.trim(),
          fixedAmount: amountNum,
          maxContributors: maxContributors ? parseInt(maxContributors, 10) : undefined,
        }),
      });
      const data = await res.json();
      if (!res.ok) {
        setFormError(data.error || "Something went wrong. Please try again.");
        setSubmitting(false);
        return;
      }
      setLocation(`/chip-in/dashboard/${data.id}`);
    } catch {
      setFormError("Unable to connect. Please check your connection and try again.");
      setSubmitting(false);
    }
  }

  if (authLoading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <Loader2 className="w-6 h-6 animate-spin text-muted-foreground" />
      </div>
    );
  }

  return (
    <div className="min-h-screen w-full px-6 py-16">
      <motion.div initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.5 }} className="w-full max-w-xl mx-auto">
        <div className="text-center mb-8">
          <p className="text-xs font-semibold tracking-widest uppercase text-primary mb-3">Chip In</p>
          <h1 className="font-serif text-3xl md:text-4xl font-medium mb-3">Start a group moment</h1>
          <p className="text-muted-foreground text-sm">Build the moment, set an amount, and share one link with everyone chipping in.</p>
        </div>

        <form onSubmit={handleSubmit} className="bg-card border border-border rounded-3xl p-6 md:p-8 shadow-sm space-y-6">
          <div className="grid grid-cols-2 gap-3">
            <div>
              <label className="block text-xs font-medium text-muted-foreground mb-1.5">Your name</label>
              <input type="text" value={organizerName} onChange={e => setOrganizerName(e.target.value)}
                placeholder="Jordan Lee"
                className="w-full h-11 px-4 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition" />
            </div>
            <div>
              <label className="block text-xs font-medium text-muted-foreground mb-1.5">Recipient's name</label>
              <input type="text" value={recipientName} onChange={e => setRecipientName(e.target.value)}
                placeholder="Maria Chen"
                className="w-full h-11 px-4 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition" />
            </div>
          </div>

          <div>
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">Recipient's phone (optional — needed before sending)</label>
            <input type="tel" value={recipientPhone} onChange={e => setRecipientPhone(e.target.value)}
              placeholder="+1 (555) 123-4567"
              className="w-full h-11 px-4 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition" />
          </div>

          <div>
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">Occasion</label>
            <select value={occasion} onChange={e => setOccasion(e.target.value)}
              className="w-full h-11 px-4 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition">
              {OCCASIONS.map(o => <option key={o} value={o}>{o}</option>)}
            </select>
          </div>

          <div>
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">Give it a title</label>
            <input type="text" value={giftTitle} onChange={e => setGiftTitle(e.target.value)}
              placeholder="Happy Birthday, Maria!" maxLength={100}
              className="w-full h-11 px-4 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition" />
          </div>

          <div>
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">Theme</label>
            <div className="grid grid-cols-4 gap-2">
              {EXPERIENCE_LIST.map(exp => (
                <button
                  key={exp.id} type="button" onClick={() => setExperience(exp.id)}
                  className={`aspect-square rounded-xl border-2 transition-all ${experience === exp.id ? "border-primary scale-95" : "border-transparent"}`}
                  style={gradientStyle(exp.id)}
                  title={exp.name}
                />
              ))}
            </div>
          </div>

          <div>
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">Personal note (for the recipient)</label>
            <textarea value={personalNote} onChange={e => setPersonalNote(e.target.value)}
              placeholder="Write something they'll remember..." rows={3}
              className="w-full px-4 py-2.5 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition resize-none" />
          </div>

          <div>
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">Video message (optional)</label>
            {!videoPreviewUrl ? (
              <button type="button" onClick={() => videoInputRef.current?.click()}
                className="w-full h-24 rounded-xl border-2 border-dashed border-border flex flex-col items-center justify-center gap-1.5 text-muted-foreground hover:border-primary/40 transition-colors">
                <Video className="w-5 h-5" />
                <span className="text-xs">Tap to add a video</span>
              </button>
            ) : (
              <div className="relative rounded-xl overflow-hidden border border-border">
                <video src={videoPreviewUrl} className="w-full h-40 object-cover" controls />
                <button type="button" onClick={handleRemoveVideo}
                  className="absolute top-2 right-2 w-7 h-7 rounded-full bg-black/60 text-white flex items-center justify-center">
                  <X className="w-4 h-4" />
                </button>
              </div>
            )}
            <input ref={videoInputRef} type="file" accept="video/*" onChange={handleVideoSelect} className="hidden" />
            {videoError && <p className="text-xs text-destructive mt-1.5">{videoError}</p>}
          </div>

          <div>
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">Photos (up to {MAX_PHOTOS})</label>
            <div className="grid grid-cols-3 gap-2">
              {photos.map(p => (
                <div key={p.id} className="relative aspect-square rounded-xl overflow-hidden border border-border">
                  <img src={p.previewUrl} alt="" className="w-full h-full object-cover" />
                  <button type="button" onClick={() => handleRemovePhoto(p.id)}
                    className="absolute top-1.5 right-1.5 w-6 h-6 rounded-full bg-black/60 text-white flex items-center justify-center">
                    <X className="w-3.5 h-3.5" />
                  </button>
                </div>
              ))}
              {photosUploading.map(p => (
                <div key={p.id} className="aspect-square rounded-xl border border-border flex items-center justify-center bg-secondary/40">
                  <Loader2 className="w-4 h-4 animate-spin text-muted-foreground" />
                </div>
              ))}
              {photos.length + photosUploading.length < MAX_PHOTOS && (
                <button type="button" onClick={() => photoInputRef.current?.click()}
                  className="aspect-square rounded-xl border-2 border-dashed border-border flex items-center justify-center text-muted-foreground hover:border-primary/40 transition-colors">
                  <Plus className="w-5 h-5" />
                </button>
              )}
            </div>
            <input ref={photoInputRef} type="file" accept="image/jpeg,image/png,image/webp,image/heic" multiple onChange={handlePhotoSelect} className="hidden" />
            {photoError && <p className="text-xs text-destructive mt-1.5">{photoError}</p>}
          </div>

          <div>
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">Song link (optional)</label>
            <input type="url" value={playlistUrl} onChange={e => setPlaylistUrl(e.target.value)}
              placeholder="https://open.spotify.com/..."
              className="w-full h-11 px-4 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition" />
          </div>

          <div className="border-t border-border pt-6">
            <label className="block text-xs font-medium text-muted-foreground mb-1.5">
              Message to contributors <span className="text-foreground">(required)</span>
            </label>
            <textarea value={pitchMessage} onChange={e => setPitchMessage(e.target.value)}
              placeholder="Putting this together for Maria's birthday — a video from all of us plus some cash so she can treat herself. She has no idea, so keep it quiet!"
              rows={3} maxLength={500}
              className="w-full px-4 py-2.5 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition resize-none" />
            <p className="text-xs text-muted-foreground mt-1">Shown to everyone before they pay — never to {recipientName || "the recipient"}.</p>
          </div>

          <div className="grid grid-cols-2 gap-3">
            <div>
              <label className="block text-xs font-medium text-muted-foreground mb-1.5">Amount per person</label>
              <input type="number" inputMode="decimal" min={10} max={1000} step={1} value={fixedAmount} onChange={e => setFixedAmount(e.target.value)}
                placeholder="30"
                className="w-full h-11 px-4 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition" />
              <p className="text-xs text-muted-foreground mt-1">$10–$1,000</p>
            </div>
            <div>
              <label className="block text-xs font-medium text-muted-foreground mb-1.5">Max contributors</label>
              <input type="number" inputMode="numeric" min={1} max={20} step={1} value={maxContributors} onChange={e => setMaxContributors(e.target.value)}
                className="w-full h-11 px-4 rounded-xl border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary/30 transition" />
              <p className="text-xs text-muted-foreground mt-1">Up to 20</p>
            </div>
          </div>

          <label className="flex items-center gap-2 text-sm cursor-pointer">
            <input type="checkbox" checked={organizerContributes} onChange={e => setOrganizerContributes(e.target.checked)} className="rounded border-border" />
            I'll chip in too
          </label>

          {formError && <p className="text-sm text-destructive">{formError}</p>}

          <Button type="submit" size="lg" className="w-full rounded-full" disabled={submitting}>
            {submitting ? <Loader2 className="w-4 h-4 animate-spin" /> : "Create the campaign"}
          </Button>
          <p className="text-xs text-center text-muted-foreground">You'll get a link to share right after — nothing is charged until people chip in.</p>
        </form>
      </motion.div>
    </div>
  );
}
